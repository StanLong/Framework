# 哈夫曼编码

哈夫曼编码(Huffman Coding)，又称霍夫曼编码，是一种编码方式，可变字长编码(VLC)的一种。Huffman于1952年提出一种编码方法，该方法完全依据字符出现概率来构造异字头的平均长度最短的码字，有时称之为最佳编码，一般就叫做Huffman编码（有时也称为霍夫曼编码）。

**哈夫曼编码，主要目的是根据使用频率来最大化节省字符（编码）的存储空间**

## 哈夫曼编码的生成过程

假如一段信息里只有A，B，C，D，E，F这6个字符，他们出现的次数依次是2次，3次，7次，9次，18次，25次，如何设计对应的编码呢？

我们不妨把这6个字符当做6个叶子结点，把字符出现次数当做结点的权重，以此来生成一颗哈夫曼树：

![](.././doc/28.png)

哈夫曼树的每一个结点包括左、右两个分支，二进制的每一位有0、1两种状态，我们可以把这两者对应起来，结点的左分支当做0，结点的右分支当做1，如下图

![](.././doc/29.png)

这样一来，从哈夫曼树的根结点到每一个叶子结点的路径，都可以等价为一段二进制编码：

![](.././doc/30.png)

上述过程借助哈夫曼树所生成的二进制编码，就是哈夫曼编码。

```java
package com.stanlong;

import lombok.Getter;
import lombok.Setter;
import lombok.ToString;

import java.util.*;

/**
 * 哈夫曼编码
 */
public class DataStructure {

    // 哈夫曼编码表， 形如 {65=11100, 66=11101} key 为字符串对应的字节， value 为 字节对应的哈夫曼树路径
    static Map<Byte, String> huffmanCode = new HashMap<>();
    // 存储某个叶子节点的路径
    static StringBuilder stringbuilder = new StringBuilder();

    public static void main(String[] args) throws Exception {
        // 要编码的字符串
        String content = "AABBBCCCCCCCDDDDDDDDDEEEEEEEEEEEEEEEEEEFFFFFFFFFFFFFFFFFFFFFFFFF";
        // 将要编码的字符串转化成Byte数组
        byte[] contentBytes = content.getBytes();
        System.out.println("原字符串: " + content);
        System.out.println("原字符串对应的字节数组: " + Arrays.toString(contentBytes));
        // 没压缩前： [65, 65, 66, 66, 66, 67, 67, 67, 67, 67, 67, 67, 68, 68, 68, 68, 68, 68, 68, 68, 68, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 69, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70, 70]

        // 对原始的字节数组进行哈夫曼压缩
        byte[] huffmanCodeBytes = huffmanZip(contentBytes);
        System.out.println("按照哈夫曼编码表压缩后的字节数组: " + Arrays.toString(huffmanCodeBytes));
        // 压缩后：[-25, 59, -34, -1, -1, -1, -2, -37, 109, -74, -86, -86, -86, -86, -96, 0, 0, 0]

        // 对压缩后的字节数组进行解码
        byte[] sourceBytes = decode(huffmanCode, huffmanCodeBytes);
        System.out.println("按哈夫曼编码表解压后的字节数组: " + Arrays.toString(sourceBytes));
        System.out.println("按哈夫曼编码表解码后的字符串: " + new String(sourceBytes)); // 少了几个F
    }

    /**
     * 哈夫曼压缩
     * @param contentBytes 原始字节数组
     * @return 哈夫曼压缩后的结果
     */
    public static byte[] huffmanZip(byte[] contentBytes){
        // 1. 将传过来的字节数组转化成节点列表
        System.out.println("将字节数组转换成节点列表");
        List<Node> nodeList = getNode(contentBytes);


        // 2. 将节点列表转化成哈夫曼树
        System.out.println("将节点列表转换成哈夫曼树");
        Node huffmanTreeRoot = createHuffmanTree(nodeList);
        System.out.println("前序遍历哈夫曼树： ");
        preOrder(huffmanTreeRoot);

        // 3. 生成哈夫曼编码
        generateHuffmanCode(huffmanTreeRoot);
        System.out.println("生成的哈夫曼编码表: " + huffmanCode);

        // 4. 根据哈夫曼编码对原始的字节数组进行压缩
        byte[] huffmanCodeBytes = zipHuffmanCode(contentBytes, huffmanCode);

        // 5. 返回压缩后的字节数组
        return huffmanCodeBytes;
    }

    /**
     * 把字节数组转换成nodeList
     * @param bytes 接受的字节数组
     * @return 返回 nodeList
     */
    public static List<Node> getNode(byte[] bytes){
        ArrayList<Node> nodeList = new ArrayList<>();

        // 遍历bytes， 统计每一个byte出现的次数
        Map<Byte, Integer> map = new HashMap<>();
        int value=0;
        for(byte key : bytes){
            // byte 作为key， byte 出现的次数作为value
            // 方式一
//            value = map.getOrDefault(key, 0);
//            value = value + 1;
//            map.put(key, value);

            // 方式二
            map.merge(key, 1, Integer::sum);
        }
        // System.out.println(map);
        // {65=2, 66=3, 67=7, 68=9, 69=18, 70=25}

        // 把map中的键值对装换成一个node对象， 并加入到 nodeList 里
        map.forEach((k, v) -> nodeList.add(new Node(k, v)));
        return nodeList;
    }

    /**
     * 创建一颗哈夫曼树
     * @param nodeList 节点列表
     * @return 哈夫曼树
     */
    public static Node createHuffmanTree(List<Node> nodeList){
        while (nodeList.size() > 1){
            // 排序
            Collections.sort(nodeList);

            // 取出第一颗最小的二叉树
            Node left = nodeList.get(0);
            // 取出第二颗最小的二叉树
            Node right = nodeList.get(1);

            // 创建新的二叉树， 新的二叉树没有数据，只有权值
            Node parent = new Node(null, left.getWeight() + right.getWeight());
            parent.setLeft(left);
            parent.setRight(right);

            nodeList.remove(left);
            nodeList.remove(right);
            nodeList.add(parent);
        }
        return nodeList.get(0);
    }

    /**
     * 前序遍历
     * @param root 前序遍历结果
     */
    public static void preOrder(Node root){
        if(root != null){
            root.preOrder();
        }else {
            System.out.println("哈夫曼树为空");
        }
    }

    /**
     * 生成哈夫曼编码
     * @param node 哈夫曼树
     * @param code 路径：往左0， 往右1
     * @param stringbuilder 拼接路径
     */
    public static void generateHuffmanCode(Node node, String code, StringBuilder stringbuilder){
        StringBuilder stringBuilder2 = new StringBuilder(stringbuilder);
        stringBuilder2.append(code);
        if(node != null){
            if(node.getValue() == null){ // 非叶子节点
                // 向左递归
                generateHuffmanCode(node.getLeft(), "0", stringBuilder2);
                // 向右递归
                generateHuffmanCode(node.getRight(), "1", stringBuilder2);
            }else { // 如果是叶子节点
                huffmanCode.put(node.getValue(), stringBuilder2.toString());
            }
        }
    }

    // 重载 generateHuffmanCode， 方便调用
    public static Map<Byte, String> generateHuffmanCode(Node node){
        if(node == null){
            return null;
        }
        generateHuffmanCode(node.getLeft(), "0", stringbuilder);
        generateHuffmanCode(node.getRight(), "1", stringbuilder);
        return huffmanCode;
    }

    /**
     * 返回按哈夫曼编码压缩后的字节数组
     * @param bytes 原始字符串数组对应的字节数组
     * @param huffmanCode 生成的哈夫曼编码表
     * @return 按哈夫曼编码压缩后对应的字节数组 11100111(补码)-> 11100111-1 = 11100110(反码)->10011001(源码)->-25
     */
    public static byte[] zipHuffmanCode(byte[] bytes, Map<Byte, String> huffmanCode){
        // 1. 将 huffmanCode 转换成哈夫曼编码字符串
        StringBuilder stringBuilder = new StringBuilder();
        for(byte b : bytes){
            stringBuilder.append(huffmanCode.get(b));
        }
        System.out.println("原字节数组对应的哈夫曼编码：(计算机中是按照补码来存储的)");
        System.out.println(stringBuilder.toString());

        // 按长度为8位进行截取， 每8位是一个byte
        int len=0;
        if(stringBuilder.length() % 8 == 0){
            len = stringBuilder.length() / 8;
        }else {
            len = stringBuilder.length() / 8 + 1;
        }
        // 或者直接写成 int len = (stringBuilder.length()+7)/8;

        // 创建存储压缩后的byte数组
        byte[] huffmanCodeBytes = new byte[len];
        int index = 0; // 记录第几个byte
        for(int i=0; i<stringBuilder.length(); i=i+8){
            String strByte = "";
            if(i + 8 > stringBuilder.length()){ // 不够8位
                strByte = stringBuilder.substring(i);
            }else{
                strByte = stringBuilder.substring(i, i+8);
            }


            // 将 strByte 转化成一个byte, 放入到 huffmanCodeBytes
            huffmanCodeBytes[index] = (byte) Integer.parseInt(strByte, 2); // 按二进制转换
            index++;
        }
        return huffmanCodeBytes;
    }

    /**
     * 将一个byte转成8位二进制的字符串
     * @param flag 是否需要补位，true 正数需要补高位
     * @param b 要转换的 byte
     * @return 转换后的二进制字符串(补码)
     */
    public static String byteToBitString(boolean flag, byte b){
        int temp = b;

        // 正数高位补0
        if(flag){
            temp |= 256; // 按位与256
        }

        String str = Integer.toBinaryString(temp); // 返回 temp 对应的二进制补码
        if(flag){
            return str.substring(str.length() - 8);
        }else{
            return str;
        }
    }

    /**
     * 按照哈夫曼编码表解码
     * @param huffmanCodes 哈夫曼编码表
     * @param huffmanBytes 按哈夫曼编码压缩后的字节数组
     * @return 对照哈夫曼编码表解码后的字节数组
     */
    public static byte[] decode(Map<Byte, String> huffmanCodes, byte[] huffmanBytes){
        // 1. 先得到 huffmanBytes 对应的二进制字符串
        StringBuilder stringBuilder = new StringBuilder();
        for(int i=0; i<huffmanBytes.length; i++){
            // 判断是不是最后一个字节
            boolean flag = (i == huffmanBytes.length - 1);
            stringBuilder.append(byteToBitString(!flag, huffmanBytes[i]));
        }
       System.out.println("压缩后的字节数组对应的二进制码：" + stringBuilder.toString());

        // 把字符串按照哈夫曼编码表进行解码
        Map<String, Byte> map = new HashMap<>();

        // 反向查找，需要将哈夫曼编码表进行调换
        huffmanCodes.forEach((k, v) -> map.put(v, k));
        System.out.println("调换后的哈夫曼编码表： " + map);

        //创建集合，存放 byte
        List<Byte> list = new ArrayList<>();
        for(int i=0; i<stringBuilder.length();){
            int count = 1;
            boolean flag = true;
            Byte b = null;

            while (flag){
                // 取出一个二进制位 1 或者 0
                String key = stringBuilder.substring(i, i+count);
                b = map.get(key);
                if(b == null){
                    count ++ ;
                }else {
                    flag = false;
                }
            }
            list.add(b);
            i = i+count;
        }
        byte b[] = new byte[list.size()];
        for(int i=0; i<b.length; i++){
            b[i] = list.get(i);
        }
        return b;
    }


}

/**
 * 节点带权值
 */
@ToString(of={"value", "weight"})
@Setter
@Getter
class Node implements Comparable<Node>{

    private Byte value; // 存放数据, byte的取值范围是-128到+127
    private int weight; // 存放数据的权值，表示数据出现的次数
    private Node left;
    private Node right;

    public Node(Byte value, int weight){
        this.value = value;
        this.weight = weight;
    }

    @Override
    public int compareTo(Node node) {
        return this.weight - node.weight; // 从小到大排序
    }

    // 前序遍历
    public void preOrder(){
        System.out.println(this);
        if(this.left != null){
            this.left.preOrder();
        }
        if(this.right != null){
            this.right.preOrder();
        }
    }
}
```
