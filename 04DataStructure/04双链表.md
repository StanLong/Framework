# 双链表

## Java版

```java
package com.stanlong;

public class DataStructure {

    public static void main(String[] args) throws Exception {

        // 准备两个链表
        // 链表1
        Node node1 = new Node(1);
        Node node2 = new Node(2);
        Node node3 = new Node(3);
        Node node4 = new Node(4);

        DoubleLinkedList dll = new DoubleLinkedList();
        dll.add(node1);
        dll.add(node2);
        dll.add(node3);
        dll.add(node4);
        dll.show();

        System.out.println("=================修改=================");

        dll.update(node4);
        dll.show();

    }

}

// 模拟内存中的节点
class Node{
    public int data; // 数据域
    public Node pre; // 指针域（指向前一个节点）
    public Node next; // 指针域（指向后一个节点）


    public Node(int data){
        this.data = data;
    }

    @Override
    public String toString() {
        return "Node[" + "data=" + data +  "]";
    }
}

// 双向链表
class DoubleLinkedList{

    private Node headNode = new Node(0);

    // 新增
    public void add(Node node){
        Node temp = headNode;
        while (true){
            if(temp.next == null){
                break;
            }
            temp = temp.next;
        }
        temp.next = node;
        node.pre = temp;
    }

    // 查看
    public void show(){
        Node temp = headNode;
        if(temp.next == null){
            System.out.println("链表是空的");
        }
        while (true){
            if(temp.next == null){
                break;
            }
            System.out.println(temp.next);
            temp = temp.next;
        }
    }
    // 修改
    public void update(Node node){
        Node temp = headNode;
        if(temp.next == null){
            System.out.println("链表是空的");
        }
        while (true){
            if(temp.next == null){
                break;
            }
            if(temp.data == node.data){
                temp.data = 100;
                break;
            }
            temp = temp.next;
        }
    }
    // 删除
    public void delete(Node node){
        Node temp = headNode;
        if(temp.next == null){
            System.out.println("没有节点可以删除，链表是空的");
        }
        while (true){
            if(temp.next == null){
                System.out.println("遍历结束，没有找到要删除的节点");
                break;
            }
            if(temp.data == node.data){
                temp.pre.next = temp.next;
                if(temp.next != null){
                    temp.next.pre = temp.pre;
                }

            }
            temp = temp.next;
        }
    }
}
```
