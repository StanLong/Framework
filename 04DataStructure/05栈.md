# 栈

## java版

用数组模拟栈

```java
package com.stanlong;

import java.util.Scanner;

/**
 * 栈
 * 特点：
 * 1. 先入后出的有序列表
 * 2. 变化的一端称为栈顶， 固定的一端为栈底
 */
public class DataStructure {

    public static void main(String[] args) throws Exception {
        ArrayStack as = new ArrayStack(4);
        String key= "";
        boolean loop = true;
        Scanner sc = new Scanner(System.in);
        while (loop){
            System.out.println("push:入栈");
            System.out.println("pop:出栈");
            System.out.println("show:展示栈");
            System.out.println("exit:退出栈");
            System.out.println("请输入选择:");

            key = sc.next();
            switch (key){
                case "show":
                    try{
                        as.show();
                    }catch (Exception e){
                        System.out.println(e.getMessage());
                    }
                    break;
                case "push":
                    System.out.println("输入入栈值:");
                    int value = sc.nextInt();
                    as.push(value);
                    break;
                case "pop":
                    try {
                        int result = as.pop();
                        System.out.printf("出栈的数据是 %d \n", result);
                    }catch (Exception e){
                        System.out.println(e.getMessage());
                    }
                    break;
                case "exit":
                    sc.close();
                    loop = false;
                    break;
                default:
                    break;
            }
        }

    }
}


/**
 * 用数组模拟栈
 */
class ArrayStack{
    private int maxSize; // 栈的大小
    private int[] stack; // 模拟栈的数组
    private int top =-1; // 栈顶，初始值为 -1

    public ArrayStack(int maxSize){
        this.maxSize = maxSize;
        stack = new int[this.maxSize];
    }

    public boolean isFull(){
        return top == maxSize -1;
    }

    public boolean isEmpty(){
        return top == -1;
    }

    // 入栈
    public void push(int value){
        if(isFull()){
            System.out.println("栈满了");
            return;
        }
        top = top + 1;
        stack[top] = value;
    }

    // 出栈
    public int pop(){
        if(isEmpty()){
            throw new RuntimeException("栈是空的");
        }
        int value = stack[top];
        top = top -1;
        return value;
    }

    // 遍历栈里的元素
    public void show(){
        if(isEmpty()){
            throw new RuntimeException("栈是空的");
        }
        for(int i=top; i>=0; i--){
            System.out.printf("stack[%d]=%d \n", i, stack[i]);
        }
    }
}
```

## Scala版

用单链表模拟栈

```scala
package com.stanlong.scala

import scala.util.control.Breaks.{break, breakable}


/**
 * 栈
 */
object ForDemo {
    def main(args: Array[String]): Unit = {
        var sls = new SingleLinkedStack
        sls.push(1)
        sls.push(2)
        sls.push(3)
        sls.push(4)
        sls.push(5)
        sls.show()
        sls.pop()
        sls.pop()


    }


    class Node(in_data:Int){
        var data = this.in_data
        var next:Node = null

        override def toString: String = {
            "Node[" + data + "] = " + data
        }
    }

    // 用单链表模拟栈
    class SingleLinkedStack{

        var headNode = new Node(0)

        // 入栈
        def push(data:Int): Unit ={
            var temp:Node = headNode
            while (temp.next !=null){
                temp = temp.next
            }
            var node = new Node(data)
            temp.next = node
        }

        // 出栈
        def pop(): Unit={
            if(headNode == null){
                println("链表是空的")
                return
            }
            var temp:Node = headNode
            // 首先 temp 节点遍历到链表最后节点的前一个节点
            breakable{
                while (true){
                    if(temp.next.next == null){
                        break
                    }
                    temp = temp.next
                }
            }
            println("出栈节点: " + temp.next.data)
            temp.next = null
        }



        def show(): Unit ={
            if(headNode.next == null){
                println("链表是空的")
                return
            }
            var temp:Node = headNode
            breakable{
                while (true){
                    if(temp.next == null){
                        break
                    }
                    println("入栈节点: " + temp.next.data)
                    temp = temp.next
                }
            }

        }
    }
}
```
