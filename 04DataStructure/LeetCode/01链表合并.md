# 链表合并

```scala
package com.stanlong.scala

import scala.util.control.Breaks.{break, breakable}

/**
 * 单向链表合并: 输入是两个有序链表，和并这两个链表后输出一个有序链表
 * 输入：
 * 链表1：head1->1->3->5->7
 * 链表2: head2->2->4->6->8
 * 输出:
 * 链表: head->1->2->3->4->5->6->7->8
 */
object ForDemo {
    def main(args: Array[String]): Unit = {
        // 首先准备两个单向链表
        val node01 = new Node(1)
        val node02 = new Node(3)

        val sll1 = new SingleLinkedList()
        sll1.add(node01)
        sll1.add(node02)
        sll1.show()
        println(sll1)


        val node05 = new Node(2)
        val node06 = new Node(4)
        val sll2 = new SingleLinkedList()
        sll2.add(node05)
        sll2.add(node06)
        sll2.show()
        println(sll2)

        println("==========有序列表合并===========")

        // sll1 和 sll2 合并
        val head = new Node(0)
        var temp = head
        var curr1 = sll1.headNode.next
        var next1:Node = null
        var curr2 = sll2.headNode.next
        var next2:Node = null
        breakable{
            while (true){
                if(curr1 == null || curr2 == null){
                    break()
                }
                if(curr1.data <= curr2.data){
                    next1 = curr1.next
                    curr1.next = temp.next
                    temp.next = curr1
                    curr1 = next1
                }else{
                    next2 = curr2.next
                    breakable{
                        while (true){
                            if(temp.next == null){
                                break()
                            }
                            temp = temp.next
                        }
                    }
                    curr2.next = temp.next
                    temp.next = curr2
                    curr2 = next2
                }
            }

        }

        breakable{
            while(true){
                if(temp == null){
                    break()
                }
                println(temp)
                temp = temp.next
            }
        }
    }

    class Node(in_data:Int){
        var data = this.in_data
        var next:Node = null

        override def toString: String = "Node[" + "data=" + data + "]"
    }

    class SingleLinkedList{
        var headNode = new Node(0)

        // 在链表尾部新增节点
        def add(node: Node): Unit ={
            var temp = headNode
            breakable{
                while (true){
                    if(temp.next == null){
                        break()
                    }
                    temp = temp.next
                }
            }
            temp.next = node

        }

        // 展示所有的节点
        def show(): Unit ={
            if(headNode.next == null){
                println("链表为空")
                return
            }
            var temp = headNode.next
            breakable{
                while(true){
                    if(temp == null){
                        break()
                    }
                    println(temp)
                    temp = temp.next
                }
            }
        }
    }
}
```
